void solve(stack<int>&inputStack,int count,int size){
	if(count==size/2){
		inputStack.pop();
		return;
	}
	int ans=inputStack.top();
	inputStack.pop();
	count++;
	solve(inputStack,count,size);
	inputStack.push(ans);
	
}
